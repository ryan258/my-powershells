# These functions allow you to run the Drupal scripts from PowerShell.
# They work by calling the scripts inside the WSL environment.

function Convert-WindowsPathToWslPath {
    param([string]$Path)
    $drive = $Path.Substring(0, 1).ToLower()
    $rest = $Path.Substring(2) -replace '\\', '/'
    return "/mnt/$drive$rest"
}

function new-drupal {
    param(
        [Parameter(Mandatory=$true, Position=0)]
        [string]$ProjectName,

        [Parameter(Position=1)]
        [string]$GitRepo,

        [ValidateSet('site', 'core', 'module', 'theme')]
        [string]$SetupType = 'site',
        [string]$Docroot,
        [switch]$SkipLaunch,
        [switch]$SkipSiteInstall,
        [switch]$LinkExtensionDependencies
    )
    $scriptPath = Join-Path $PSScriptRoot "new-drupal-project.ps1"
    $wslScriptPath = Convert-WindowsPathToWslPath -Path $scriptPath
    
    $wslArgs = "-ProjectName $ProjectName"
    if ($PSBoundParameters.ContainsKey('GitRepo')) { $wslArgs += " -GitRepo $GitRepo" }
    if ($PSBoundParameters.ContainsKey('SetupType')) { $wslArgs += " -SetupType $SetupType" }
    if ($PSBoundParameters.ContainsKey('Docroot')) { $wslArgs += " -Docroot $Docroot" }
    if ($PSBoundParameters.ContainsKey('SkipLaunch')) { $wslArgs += " -SkipLaunch" }
    if ($PSBoundParameters.ContainsKey('SkipSiteInstall')) { $wslArgs += " -SkipSiteInstall" }
    if ($PSBoundParameters.ContainsKey('LinkExtensionDependencies')) { $wslArgs += " -LinkExtensionDependencies" }

    wsl.exe -- pwsh -File $wslScriptPath $wslArgs
}

function new-drupal-core {
    $scriptPath = Join-Path $PSScriptRoot "drupal-core-dev.ps1"
    $wslScriptPath = Convert-WindowsPathToWslPath -Path $scriptPath
    wsl.exe -- pwsh -File $wslScriptPath
}
